Register & Login
open /sahana
clickAndWait link=Register
click t2_person_name
storeRandom 8 user
storeRandom 8 domain
store ${user}@${domain}.com email
type t2_person_name ${user}
type t2_person_email ${email}
storeRandom 8 password
type t2_person_password ${password}
clickAndWait //input[@value='Submit']
verifyTextPresent You have been successfully registered
open /sahana/default/login
clickAndWait link=Logout
verifyTextPresent Logged Out
open /sahana/default/login
clickAndWait link=Login
type t2_person_email ${email}
type t2_person_password ${password}
clickAndWait //input[@value='Submit']
verifyTextPresent Logged In
